// Bloquer le défilement de la page lorsqu'on utilise les flèches du clavier window.addEventListener('keydown', (event) => { // Empêcher le défilement avec les touches fléchées if (event.key === 'ArrowUp' || event.key === 'ArrowDown' || event.key === 'ArrowLeft' || event.key === 'ArrowRight') { event.preventDefault(); } // Déplacement en continu avec les touches directionnelles du clavier if (moveInterval) return; // Si un mouvement est déjà en cours, ne pas répéter switch (event.key) { case 'ArrowUp': moveInterval = setInterval(() => moveCharacter('up'), 50); break; case 'ArrowDown': moveInterval = setInterval(() => moveCharacter('down'), 50); break; case 'ArrowLeft': moveInterval = setInterval(() => moveCharacter('left'), 50); break; case 'ArrowRight': moveInterval = setInterval(() => moveCharacter('right'), 50); break; } }); // Bloquer le défilement de la page avec la molette de la souris window.addEventListener('wheel', (event) => { event.preventDefault(); // Empêche le défilement de la page }, { passive: false });
